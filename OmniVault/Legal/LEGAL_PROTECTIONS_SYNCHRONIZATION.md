        uint256 timestamp;       // Blockchain timestamp
        string documentType;     // Contract, Agreement, etc.
        bool verified;           // Verification status
        string jurisdiction;     // Legal jurisdiction
    }
    
    mapping(bytes32 => LegalDocument) public legalDocuments;
    mapping(address => bytes32[]) public ownerDocuments;
    
    event DocumentAnchored(
        bytes32 indexed documentId,
        string documentCID,
        address indexed owner,
        uint256 timestamp,
        string documentType
    );
    
    event DocumentVerified(
        bytes32 indexed documentId,
        address indexed verifier,
        uint256 timestamp
    );
    
    function anchorLegalDocument(
        string memory _documentCID,
        bytes32 _documentHash,
        string memory _documentType,
        string memory _jurisdiction
    ) public returns (bytes32) {
        bytes32 documentId = keccak256(
            abi.encodePacked(_documentCID, msg.sender, block.timestamp)
        );
        
        require(
            legalDocuments[documentId].timestamp == 0,
            "Document already anchored"
        );
        
        legalDocuments[documentId] = LegalDocument({
            documentCID: _documentCID,
            documentHash: _documentHash,
            owner: msg.sender,
            timestamp: block.timestamp,
            documentType: _documentType,
            verified: true,
            jurisdiction: _jurisdiction
        });
        
        ownerDocuments[msg.sender].push(documentId);
        
        emit DocumentAnchored(
            documentId,
            _documentCID,
            msg.sender,
            block.timestamp,
            _documentType
        );
        
        return documentId;
    }
    
    function verifyDocument(bytes32 _documentId) 
        public 
        view 
        returns (LegalDocument memory) 
    {
        require(
            legalDocuments[_documentId].timestamp > 0,
            "Document not found"
        );
        return legalDocuments[_documentId];
    }
    
    function getOwnerDocuments(address _owner) 
        public 
        view 
        returns (bytes32[] memory) 
    {
        return ownerDocuments[_owner];
    }
}

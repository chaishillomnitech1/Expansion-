```solidity
pragma solidity ^0.8.17;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/access/AccessControl.sol";

contract BlessingCoin is ERC20, Pausable, AccessControl {
    bytes32 public constant MINTER_ROLE = keccak256("MINTER_ROLE");
    bytes32 public constant BURNER_ROLE = keccak256("BURNER_ROLE");
    
    uint256 public constant MAX_SUPPLY = 1_000_000_000 * 10**18;
    uint256 public constant TRANSACTION_FEE = 100; // 1% (10000 = 100%)
    
    address public charityWallet;
    address public treasuryWallet;
    
    mapping(address => bool) public isExemptFromFees;
    
    event CharityDonation(address indexed donor, uint256 amount, uint256 timestamp);
    event TokensBurned(address indexed burner, uint256 amount);
    
    constructor(
        address _charityWallet,
        address _treasuryWallet
    ) ERC20("BlessingCoin", "BLESS") {
        require(_charityWallet != address(0), "Invalid charity wallet");
        require(_treasuryWallet != address(0), "Invalid treasury wallet");
        
        charityWallet = _charityWallet;
        treasuryWallet = _treasuryWallet;
        
        _grantRole(DEFAULT_ADMIN_ROLE, msg.sender);
        _grantRole(MINTER_ROLE, msg.sender);
        
        // Initial minting according to tokenomics
        _mint(msg.sender, 400_000_000 * 10**18); // NFT staking
        _mint(_charityWallet, 280_000_000 * 10**18); // Zakat + charity reserve
        _mint(_treasuryWallet, 150_000_000 * 10**18); // Community treasury
        _mint(msg.sender, 100_000_000 * 10**18); // Team (will be vested)
        _mint(msg.sender, 70_000_000 * 10**18); // Liquidity
    }
    
    function transfer(address to, uint256 amount) 
        public 
        virtual 
        override 
        returns (bool) 
    {
        uint256 fee = 0;
        
        if (!isExemptFromFees[msg.sender] && !isExemptFromFees[to]) {
            fee = (amount * TRANSACTION_FEE) / 10000;
            _burn(msg.sender, fee); // Deflationary burn
        }
        
        return super.transfer(to, amount - fee);
    }
    
    function donateToCharity(uint256 amount) public {
        require(amount > 0, "Amount must be greater than 0");
        require(balanceOf(msg.sender) >= amount, "Insufficient balance");
        
        // 5% burn on charity donations
        uint256 burnAmount = (amount * 500) / 10000;
        _burn(msg.sender, burnAmount);
        
        // Transfer remaining to charity wallet
        _transfer(msg.sender, charityWallet, amount - burnAmount);
        
        emit CharityDonation(msg.sender, amount, block.timestamp);
    }
    
    function mint(address to, uint256 amount) 
        public 
        onlyRole(MINTER_ROLE) 
    {
        require(totalSupply() + amount <= MAX_SUPPLY, "Exceeds max supply");
        _mint(to, amount);
    }
    
    function burn(uint256 amount) public {
        _burn(msg.sender, amount);
        emit TokensBurned(msg.sender, amount);
    }
    
    function setExemptFromFees(address account, bool exempt) 
        public 
        onlyRole(DEFAULT_ADMIN_ROLE) 
    {
        isExemptFromFees[account] = exempt;
    }
    
    function pause() public onlyRole(DEFAULT_ADMIN_ROLE) {
        _pause();
    }
    
    function unpause() public onlyRole(DEFAULT_ADMIN_ROLE) {
        _unpause();
    }
}
```

### Zakat Smart Contract:

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.17;

import "./BlessingCoin.sol";

contract ZakatDistribution {
    BlessingCoin public blessingCoin;
    
    uint256 public constant ZAKAT_RATE = 250; // 2.5% (10000 = 100%)
    uint256 public constant LUNAR_YEAR = 354 days;
    uint256 public nisabThreshold; // In BLESS tokens
    
    struct Holder {
        uint256 lastZakatDate;
        uint256 totalPaid;
        bool autoZakatEnabled;
    }
    
    struct Recipient {
        address wallet;
        string category; // As per 8 categories
        bool verified;
        uint256 totalReceived;
    }
    
    mapping(address => Holder) public holders;
    mapping(address => Recipient) public recipients;
    address[] public recipientList;
    
    event ZakatPaid(address indexed payer, uint256 amount, uint256 timestamp);
    event ZakatDistributed(address indexed recipient, uint256 amount, string category);
    event RecipientVerified(address indexed recipient, string category);
    
    constructor(address _blessingCoin, uint256 _nisabThreshold) {
        blessingCoin = BlessingCoin(_blessingCoin);
        nisabThreshold = _nisabThreshold;
    }
    
    function payZakat() public {
        uint256 balance = blessingCoin.balanceOf(msg.sender);
        require(balance >= nisabThreshold, "Balance below Nisab threshold");
        
        Holder storage holder = holders[msg.sender];
        require(
            block.timestamp >= holder.lastZakatDate + LUNAR_YEAR,
            "Zakat not yet due"
        );
        
        uint256 zakatAmount = (balance * ZAKAT_RATE) / 10000;
        
        // Transfer Zakat to contract for distribution
        require(
            blessingCoin.transferFrom(msg.sender, address(this), zakatAmount),
            "Transfer failed"
        );
        
        holder.lastZakatDate = block.timestamp;
        holder.totalPaid += zakatAmount;
        
        emit ZakatPaid(msg.sender, zakatAmount, block.timestamp);
        
        // Distribute to verified recipients
        _distributeZakat(zakatAmount);
    }
    
    function enableAutoZakat() public {
        holders[msg.sender].autoZakatEnabled = true;
    }
    
    function _distributeZakat(uint256 amount) private {
        uint256 perRecipient = amount / recipientList.length;
        
        for (uint i = 0; i < recipientList.length; i++) {
            address recipientAddr = recipientList[i];
            Recipient storage recipient = recipients[recipientAddr];
            
            if (recipient.verified) {
                blessingCoin.transfer(recipientAddr, perRecipient);
                recipient.totalReceived += perRecipient;
                
                emit ZakatDistributed(
                    recipientAddr,
                    perRecipient,
                    recipient.category
                );
            }
        }
    }
    
    function addRecipient(
        address _wallet,
        string memory _category
    ) public {
        // Only admin can add (role management omitted for brevity)
        require(!recipients[_wallet].verified, "Already added");
        
        recipients[_wallet] = Recipient({
            wallet: _wallet,
            category: _category,
            verified: true,
            totalReceived: 0
        });
        
        recipientList.push(_wallet);
        
        emit RecipientVerified(_wallet, _category);
    }
    
    function checkZakatDue(address holder) public view returns (bool, uint256) {
        uint256 balance = blessingCoin.balanceOf(holder);
        
        if (balance < nisabThreshold) {
            return (false, 0);
        }
        
        uint256 lastPaid = holders[holder].lastZakatDate;
        bool isDue = block.timestamp >= lastPaid + LUNAR_YEAR;
        uint256 amount = (balance * ZAKAT_RATE) / 10000;
        
        return (isDue, amount);
    }
}
```

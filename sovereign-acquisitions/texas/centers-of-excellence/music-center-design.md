        uint256 lastDistribution;
    }
    
    mapping(uint256 => MusicAsset) public musicAssets;
    mapping(string => uint256[]) public genreNFTs;

    event MusicNFTMinted(

        address indexed creator,
        string genre,
        uint256 timestamp

    event ZakatCollected(
        uint256 amount,
        uint256 tokenId,
        uint256 timestamp
    );

        uint256 amount,
        address indexed beneficiary,
        string program,
        uint256 timestamp
    );

        scrollCoinContract = _scrollCoin;
    }
    
    function mintMusicNFT(
        address recipient,
        string memory ipfsHash,
        string memory metadata,
        string memory genre,
        address[] memory royaltyRecipients,

        bool isLive
    ) public onlyOwner returns (uint256) {
        // NFT minting logic with enhanced metadata

    }

        uint256 zakatAmount = (msg.value * ZAKAT_PERCENTAGE) / 10000;
        payable(zakatWallet).transfer(zakatAmount);
        
        // Update zakat statistics
        zakatStats.totalCollected += zakatAmount;
        
        // Process purchase
        // Distribute royalties
        // Emit events
        
        emit ZakatCollected(zakatAmount, tokenId, block.timestamp);
    }
    
    function distributeZakat(
        address beneficiary,
        uint256 amount,
        string memory program

        require(amount <= address(zakatWallet).balance, "Insufficient funds");
        
        // Transfer to beneficiary
        // Update statistics
        
        zakatStats.distributedAmount += amount;
        zakatStats.beneficiaryCount++;
        zakatStats.lastDistribution = block.timestamp;
        
        emit ZakatDistributed(amount, beneficiary, program, block.timestamp);
    }
    
    function getZakatStats() public view returns (ZakatDistribution memory) {
        return zakatStats;
    }
